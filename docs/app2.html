for col in numerics:
    
    if col == 'income':
        fr.loc[fr[col]<=0,col] = 0.01
        fr[col] = np.log(fr[col])

    s = fr[col].std()

    m = fr[col].mean()

    cut_level = [
        m-2*s,
        m-s,
        m+s,
        m+2*s
    ]

    # cut_level = list(np.percentile(fr[col],bounds))
    
    cut_level = [-np.inf] + cut_level + [np.inf]

    print(col)

    print(cut_level)

    fr[col] = pd.cut(
        fr[col],
        bins=cut_level,
        labels=["L","LM","M","HM","H"]
    )

    fr[col] = fr[col].astype('category')
